server {
    server_name heartly.uz www.heartly.uz;
    client_max_body_size 50M;



    location /old/ {
        limit_except GET HEAD OPTIONS {
           deny all;
        }
        # rewrite ^/old/(.*)$ /$1 break;
        proxy_pass http://127.0.0.1:8001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-Prefix /old;
    }


    location / {
        proxy_pass         http://127.0.0.1:8000/;
        proxy_http_version 1.1;
        proxy_set_header   Host $host;
        proxy_set_header   X-Real-IP $remote_addr;
        proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/heartly.uz/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/heartly.uz/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}

server {
    if ($host = www.heartly.uz) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = heartly.uz) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name heartly.uz www.heartly.uz;
    return 404; # managed by Certbot




}
